#+begin_src python
from langchain_text_splitters import MarkdownHeaderTextSplitter

splitter = MarkdownHeaderTextSplitter([
    ("#", "H1"),
    ("##", "H2"),
    ("###", "H3"),
    ("####", "H4"),
    ("#####", "H5"),
    ],
    strip_headers=False)

with open("data/md/dl.md") as f:
    chunks = splitter.split_text(f.read())

print(f'# of chunks: {len(chunks)}')
for i in range(min(len(chunks), 30)):
    print(f'== chunk {i} size: {len(chunks[i].page_content)}')
    print(chunks[i].metadata)
#+end_src
